int comment(void)//打印商品评价 
{

		printf("|%-5d  \t\t  %-20s\t\t%-20s       %-20s         %-50s  %-5c         %-6d      |\n",p->num,p->type,p->judgeMent,p->name,p->com,p->attitude,p->timeWait);
